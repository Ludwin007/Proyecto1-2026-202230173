CREATE DATABASE IF NOT EXISTS agencia_viajes;
USE agencia_viajes;

CREATE TABLE roles (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL
);

CREATE TABLE usuarios (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre_usuario VARCHAR(100) NOT NULL UNIQUE,
    contrasena VARCHAR(255) NOT NULL,
    id_rol INT NOT NULL,
    activo TINYINT(1) DEFAULT 1,
    FOREIGN KEY (id_rol) REFERENCES roles(id)
);

CREATE TABLE destinos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(150) NOT NULL UNIQUE,
    pais VARCHAR(100) NOT NULL,
    descripcion TEXT,
    clima VARCHAR(200),
    imagen_url VARCHAR(500)
);

CREATE TABLE proveedores (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(150) NOT NULL UNIQUE,
    tipo_servicio INT NOT NULL,
    pais_operacion VARCHAR(100),
    contacto VARCHAR(255)
);

CREATE TABLE paquetes (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(200) NOT NULL UNIQUE,
    id_destino INT NOT NULL,
    duracion_dias INT NOT NULL,
    descripcion TEXT,
    precio_venta DECIMAL(10,2) NOT NULL,
    capacidad_maxima INT NOT NULL,
    activo TINYINT(1) DEFAULT 1,
    FOREIGN KEY (id_destino) REFERENCES destinos(id)
);

CREATE TABLE servicios_paquete (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_paquete INT NOT NULL,
    id_proveedor INT NOT NULL,
    descripcion VARCHAR(300),
    costo_proveedor DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_paquete) REFERENCES paquetes(id),
    FOREIGN KEY (id_proveedor) REFERENCES proveedores(id)
);

CREATE TABLE clientes (
    id INT PRIMARY KEY AUTO_INCREMENT,
    dpi_pasaporte VARCHAR(50) NOT NULL UNIQUE,
    nombre_completo VARCHAR(200) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    telefono VARCHAR(30),
    correo VARCHAR(150),
    nacionalidad VARCHAR(100)
);

CREATE TABLE reservaciones (
    id INT PRIMARY KEY AUTO_INCREMENT,
    numero_reservacion VARCHAR(20) NOT NULL UNIQUE,
    fecha_creacion DATE NOT NULL,
    fecha_viaje DATE NOT NULL,
    id_paquete INT NOT NULL,
    cantidad_pasajeros INT NOT NULL,
    id_agente INT NOT NULL,
    costo_total DECIMAL(10,2) NOT NULL,
    estado VARCHAR(20) NOT NULL DEFAULT 'Pendiente',
    FOREIGN KEY (id_paquete) REFERENCES paquetes(id),
    FOREIGN KEY (id_agente) REFERENCES usuarios(id)
);

CREATE TABLE reservacion_pasajeros (
    id_reservacion INT NOT NULL,
    id_cliente INT NOT NULL,
    PRIMARY KEY (id_reservacion, id_cliente),
    FOREIGN KEY (id_reservacion) REFERENCES reservaciones(id),
    FOREIGN KEY (id_cliente) REFERENCES clientes(id)
);

CREATE TABLE pagos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_reservacion INT NOT NULL,
    monto DECIMAL(10,2) NOT NULL,
    metodo_pago INT NOT NULL,
    fecha_pago DATE NOT NULL,
    FOREIGN KEY (id_reservacion) REFERENCES reservaciones(id)
);

CREATE TABLE cancelaciones (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_reservacion INT NOT NULL UNIQUE,
    fecha_cancelacion DATE NOT NULL,
    monto_reembolsado DECIMAL(10,2) NOT NULL,
    perdida_agencia DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_reservacion) REFERENCES reservaciones(id)
);
